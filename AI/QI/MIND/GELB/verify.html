async function NC_VERIFY(files){ const out = []; for(const f of files){ const res = await fetch(f).then(r => r.ok ? r.text() : null).catch(()=>null); const raw = res || ""; const ok = !!res; // PX decode const decoded = [...raw].map(s => Engine.PX[s] || s); // MIND detect const mind = raw.match(/[○↗□⊕△◇↘⊗⚑⚡∞]+/g); // NC36 detect const nc36 = Engine.NC36.flat().filter(s => raw.includes(s)); // MOVE detect const move = raw.includes("ERR") ? Engine.MOVE.error : raw.includes("FIX") ? Engine.MOVE.borg : raw.includes("FLOW") ? Engine.MOVE["8472"] : "OK"; // STABIL detect const stabil = raw.includes("FIX") ? Engine.STABIL.borg : raw.includes("FLOW") ? Engine.STABIL["8472"] : { line:"UNBEKANNT", form:"UNBEKANNT", axis:"UNBEKANNT" }; // GEO vector const geo = { NC2: raw.length, D6: raw.split("\n").length, E12: raw.includes("ERR") ? "kritisch" : "neutral" }; out.push({ file: f, ok, raw, decoded, mind, nc36, move, stabil, geo }); } return out; }
